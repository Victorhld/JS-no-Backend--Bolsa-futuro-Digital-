//pede os valores A,B,C para o Usuário
let a =parseInt(prompt("Informe o valor de A:"))
let b =parseInt(prompt("Informe o valor de B:"))
let c =parseInt(prompt("Informe o valor de C:"))

//calcular o Delta
let delta = (b * b) - 4 * a * c

// verifica se tem raizes
if (delta >= 0){
    //se o delta for positivo ou zero,calcula as duas raizes
    let raiz1 = (-b + Math.sqrt(delta)) / (2 * a)
    let raiz2 = (-b - Math.sqrt(delta)) / (2 * a)

    //Margem o resultado para o Usuário
    alert("A primeira raiz é: "+ raiz1)
    alert("A segunda raiz é: "+ raiz2)
    alert("A segunda delta é: "+ delta)
} else {
    // se o delta for negativo, não tem raizes reais
    alert("Não há raizes reais para esta equação. (Delta Negativo)")
}

//Define a função"converterFahrenheit" que recebe uma temperatura em F
function converterFahrenheit(temperaturaFahrenheit){
    // a fórmula da conversão
    const temperaturaCelsius = (temperaturaFahrenheit - 32)*5 / 9
    // A função retorna o resultado do cálculo
return temperaturaCelsius
}
//Pede ao usuário para digitar a temperatura em Fahrenheit
const temperaturaF = prompt("Qual a sua temperatura em °F (Fahrenheit)?")

//chama a função e armazena o resultado em uma variável
const temperaturaC = converterFahrenheit(temperaturaF)

//Mostra o resultado final com 2 casas decimais
console.log('Sua temperatura em °C (Celsius) é' + temperaturaC.toFixed(2))
